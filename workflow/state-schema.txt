# State log schema (append-only, key=value)
#
# - Stored in project folder: output/<topic>_<timestamp>/state.txt
# - Each block is a snapshot of the latest state.
# - Blocks are separated by a line containing only "---".
# - Resume by reading the last block.
# - Rollback by copying a previous block and appending it as the newest.
#
# Required keys (minimum):
#   timestamp=ISO8601
#   job_id=JOB_YYYY-MM-DD_0001
#   topic=string
#   status=INIT|RESEARCH|STORY|SCRIPT|VIDEO|QA|DONE
#
# Optional keys:
#   runtime.stage=init|assets|render|done|shorts|...
#   runtime.render.status=started|success|failed
#   immersive.experience=cinematic_story|cloud_island_walk|ride_action_boat  # ride_action_boat は legacy alias
#   gate.research_review=required|optional|skipped
#   gate.story_review=required|optional|skipped
#   gate.image_prompt_review=required|optional|skipped
#   gate.video_review=required|optional|skipped
#   gate.hybridization_review=required|optional|skipped
#   review.hybridization.status=pending|approved|rejected
#   review.hybridization.note=string
#   review.hybridization.at=ISO8601
#   review.image_prompt.status=pending|approved|changes_requested
#   review.image_prompt.note=string
#   review.image_prompt.at=ISO8601
#   artifact.research=output/<topic>_<timestamp>/research.md
#   artifact.story=output/<topic>_<timestamp>/story.md
#   artifact.visual_value=output/<topic>_<timestamp>/visual_value.md
#   artifact.script=output/<topic>_<timestamp>/script.md
#   artifact.video=output/<topic>_<timestamp>/video.mp4
#   artifact.video.short.01=output/<topic>_<timestamp>/shorts/short01.mp4
#   review.video.status=pending|approved|changes_requested
#   review.video.note=string
#   review.video.at=ISO8601
#   eval.research.score=0.0-1.0
#   eval.story.score=0.0-1.0
#   eval.script.score=0.0-1.0
#   eval.manifest.score=0.0-1.0
#   selection.story.candidate_count=2
#   selection.story.chosen_id=A
#   artifact.eval_report=output/<topic>_<timestamp>/eval_report.json
#   qa.accuracy_score=0.0-1.0
#   qa.engagement_score=0.0-1.0
#   qa.consistency_score=0.0-1.0
#   runtime.attempts.research=0
#   runtime.attempts.story=0
#   runtime.attempts.script=0
#   runtime.attempts.video=0
#   runtime.scene.1.status=draft|review|revise|accepted|failed
#   runtime.scene.1.attempts=0
#   last_error=string|null
#
# Example:
#
# timestamp=2026-01-17T12:00:00Z
# job_id=JOB_2026-01-17_0001
# topic=桃太郎
# status=RESEARCH
# gate.research_review=required
# artifact.research=output/momotaro_20260117_1200/research.md
# eval.research.score=0.8500
# artifact.eval_report=output/momotaro_20260117_1200/eval_report.json
# runtime.attempts.research=1
# ---
